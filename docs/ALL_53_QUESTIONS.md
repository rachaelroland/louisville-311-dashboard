## DATABASE ACCESS

```sql
-- View all questions
SELECT * FROM l311_approved_questions
ORDER BY category, id;

-- Questions by category
SELECT category, COUNT(*)
FROM l311_approved_questions
GROUP BY category
ORDER BY COUNT(*) DESC;

-- High priority questions
SELECT question_text, typical_urgency
FROM l311_approved_questions
WHERE typical_urgency = 'high';

-- Most recent additions
SELECT question_text, created_at
FROM l311_approved_questions
ORDER BY created_at DESC
LIMIT 30;
```
